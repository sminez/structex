19/// A compiled structural regular expression backed by an underlying regular expression engine.
20///
21/// A `Structex` can be used to search for tagged substrings within a haystack supported by the
22/// regular expression engine it is backed by. The primary API for making use of a `Structex` is
23/// the [Structex::iter_tagged_captures] method which will iterate over the [TaggedCaptures] within
24/// a given haystack as it is searched.
25#[derive(Clone)]
26pub struct Structex<R>
27where
28    R: Re,
29{
30    raw: Arc<str>,
31    inner: Arc<Inner<R>>,
32}

52impl<R> Structex<R>
53where

56    /// Compiles a structural regular expression. Once compiled it may be used repeatedly and cloned
57    /// cheaply, but note that compilation can be an expensive process so [Structex] instances
58    /// should be reused wherever possible.
59    ///
60    /// To configure how given `Structex` is compiled, see [StructexBuilder].
61    ///
62    /// # Error
63    /// If an invalid expression is given then an error is returned. The exact expressions that are
64    /// valid to compile will depend on the underlying regular expression engine being used.
65    ///
66    /// # Example
67    /// ```
68    /// // A Structex backed by the regex crate
69    /// type Structex = structex::Structex<regex::Regex>;
70    ///
71    /// // An empty expression is always invalid
72    /// assert!(Structex::new("").is_err());
73    ///
74    /// // The top level expression must not be a bare action
75    /// assert!(Structex::new("P/I am invalid/").is_err());
76    ///
77    /// // A valid expression with a named action
78    /// assert!(Structex::new("x/hello, (world|sailor)!/ p").is_ok());
79    /// ```
80    pub fn new(se: &str) -> Result<Self, Error> {
81        StructexBuilder::new(se).build()
82    }
83
84    /// Returns the original string of this structex.
85    ///
86    /// # Example
87    /// ```
88    /// type Structex = structex::Structex<regex::Regex>;
89    ///
90    /// let se = Structex::new("x/foo.*bar/ p").unwrap();
91    /// assert_eq!(se.as_str(), "x/foo.*bar/ p");
92    /// ```
93    pub fn as_str(&self) -> &str {
94        &self.raw
95    }
96
97    /// Returns the registered [actions][Action] that were parsed from the compiled expression.
98    ///
99    /// # Example
100    /// ```
101    /// use structex::Action;
102    ///
103    /// type Structex = structex::Structex<regex::Regex>;
104    ///
105    /// let se = Structex::new("x/foo.*bar/ { p; a/baz/; }").unwrap();
106    /// let actions = se.actions();
107    ///
108    /// assert_eq!(actions.len(), 2);
109    /// assert_eq!(actions[0].as_ref(), &Action { tag: 'p', arg: None });
110    /// assert_eq!(actions[1].as_ref(), &Action { tag: 'a', arg: Some("baz".to_string()) });
111    /// ```
112    pub fn actions(&self) -> &[Arc<Action>] {
113        &self.inner.actions
114    }
115
116    /// Returns the registered tags that were parsed from the compiled expression.
117    ///
118    /// # Example
119    /// ```
120    /// type Structex = structex::Structex<regex::Regex>;
121    ///
122    /// let se = Structex::new("x/foo.*bar/ { p; a/baz/; }").unwrap();
123    /// assert_eq!(se.tags(), &['a', 'p']);
124    /// ```
125    pub fn tags(&self) -> &[char] {
126        &self.inner.tags
127    }

129    /// Iterate over all [TaggedCaptures] within the given haystack in order.
130    ///
131    /// # Examples
132    ///
133    /// By default, matches will be emitted without an associated action attached to them,
134    /// allowing you to write simple expressions that filter and refine regions of the haystack to
135    /// locate the structure you are looking for.
136    /// ```
137    /// type Structex = structex::Structex<regex::Regex>;
138    ///
139    /// let se = Structex::new(r#"
140    ///   x/(.|\n)*?\./   # split into sentences
141    ///   g/Alice/        # if the sentence contains "Alice"
142    ///   n/(\w+)\./      # extract the last word of the sentence
143    /// "#).unwrap();
144    ///
145    /// let haystack = r#"This is a multi-line
146    /// string that mentions peoples names.
147    /// People like Alice and Bob. People
148    /// like Claire and David, but really
149    /// we're here to talk about Alice.
150    /// Alice is everyone's friend."#;
151    ///
152    /// let last_words: Vec<&str> = se
153    ///     .iter_tagged_captures(haystack)
154    ///     .map(|m| m.submatch_text(1).unwrap())
155    ///     .collect();
156    ///
157    /// assert_eq!(&last_words, &["Bob", "Alice", "friend"]);
158    /// ```
159    ///
160    /// When writing more complex expressions you will want to assign tagged actions to each
161    /// matching branch in order to distinguish them:
162    /// ```
163    /// type Structex = structex::Structex<regex::Regex>;
164    ///
165    /// let se = Structex::new(r#"
166    ///   ## split into sentences
167    ///   x/(.|\n)*?\./ {
168    ///     ## if the sentence contains "Alice" extract the last word of the sentence
169    ///     g/Alice/ n/(\w+)\./ A;
170    ///     ## if it doesn't, extract the first word of the sentence
171    ///     v/Alice/ n/(\w+)/ B;
172    ///   }
173    /// "#).unwrap();
174    ///
175    /// let haystack = r#"This is a multi-line
176    /// string that mentions peoples names.
177    /// People like Alice and Bob. People
178    /// like Claire and David, but really
179    /// we're here to talk about Alice.
180    /// Alice is everyone's friend."#;
181    ///
182    /// let words: Vec<(char, &str)> = se
183    ///     .iter_tagged_captures(haystack)
184    ///     .map(|m| (m.tag().unwrap(), m.submatch_text(1).unwrap()))
185    ///     .collect();
186    ///
187    /// assert_eq!(
188    ///     &words,
189    ///     &[('B', "This"), ('A', "Bob"), ('A', "Alice"), ('A', "friend")]
190    /// );
191    /// ```
192    pub fn iter_tagged_captures<'h>(
193        &'h self,
194        haystack: &'h R::Haystack,
195    ) -> TaggedCapturesIter<'h, R> {
196        TaggedCapturesIter::new(&self.inner.inst, self.inner.clone(), haystack)
197    }
198}

211/// A configurable builder for a [Structex].
212///
213/// This builder can be used to configure which action tags are allowed to be used within the
214/// compiled expression as well as set how action argument strings should be processed.
215#[derive(Clone)]
216pub struct StructexBuilder {
217    expr: String,
218    action_arg_fn: Arc<dyn ActionArgFn>,
219    require_actions: bool,
220    allowed_argless_tags: Option<String>,
221    allowed_single_arg_tags: Option<String>,
222}
223
224impl StructexBuilder {
225    /// Constructs a new builder with the default configuration for the given expression.
226    ///
227    /// If the pattern is invalid then an error will be returned when [StructexBuilder::build] is
228    /// called.
229    pub fn new(expr: impl Into<String>) -> Self {
230        Self {
231            expr: expr.into(),
232            action_arg_fn: Arc::new(newline_and_tab_string),
233            require_actions: false,
234            allowed_argless_tags: None,
235            allowed_single_arg_tags: None,
236        }
237    }
238
239    /// This prevents interpretation of `\n` and `\t` escape sequences inside of action argument
240    /// strings (the default behaviour).
241    ///
242    /// # Example
243    /// ```
244    /// use structex::{Structex, StructexBuilder};
245    ///
246    /// let expr = "x/.*foo/ p/\nfound foo\t/";
247    /// let se: Structex<regex::Regex> = StructexBuilder::new(expr)
248    ///     .raw_arg_strings()
249    ///     .build()
250    ///     .unwrap();
251    ///
252    /// assert_eq!(se.actions()[0].arg.as_deref(), Some("\nfound foo\t"));
253    /// ```
254    pub fn raw_arg_strings(mut self) -> Self {
255        self.action_arg_fn = Arc::new(raw_arg_string);
256        self
257    }
258
259    /// This sets a custom argument mapping function that will be called for all tag arguments
260    /// found within the compiled expression.
261    ///
262    /// # Example
263    /// ```
264    /// use structex::{Structex, StructexBuilder};
265    ///
266    /// let expr = "x/.*foo/ p/found foo/";
267    /// let se: Structex<regex::Regex> = StructexBuilder::new(expr)
268    ///     .action_argument_fn(|s| s.to_ascii_uppercase())
269    ///     .build()
270    ///     .unwrap();
271    ///
272    /// assert_eq!(se.actions()[0].arg.as_deref(), Some("FOUND FOO"));
273    /// ```
274    pub fn action_argument_fn<F>(mut self, f: F) -> Self
275    where
276        F: Fn(String) -> String + 'static,
277    {
278        self.action_arg_fn = Arc::new(f);
279        self
280    }
281
282    /// Require all expression chains to end in an [Action].
283    ///
284    /// By default it is permitted to omit specifing an action tag at the end of expression chains
285    /// which will result in simply emitting the match itself. When calling this method such
286    /// branches will result in compilation errors instead.
287    ///
288    /// # Example
289    /// ```
290    /// use structex::{Structex, StructexBuilder};
291    ///
292    /// let expr = "x/.*foo/";
293    ///
294    /// // By default, actions are not required.
295    /// assert!(Structex::<regex::Regex>::new(expr).is_ok());
296    ///
297    /// // If actions are marked as required, the above expression will produce a compilation error.
298    /// assert!(
299    ///     StructexBuilder::new(expr)
300    ///         .require_actions()
301    ///         .build::<regex::Regex>()
302    ///         .is_err()
303    /// );
304    /// ```
305    pub fn require_actions(mut self) -> Self {
306        self.require_actions = true;
307        self
308    }
309
310    /// This sets the allowed tags when no slash delimited argument is provided.
311    ///
312    /// By default, all tags are allowed but this may be used to cause a compile error if the input
313    /// expression contains tags that were not expected.
314    ///
315    /// # Example
316    /// ```
317    /// use structex::{Structex, StructexBuilder};
318    ///
319    /// let expr = "x/.*foo/ A";
320    ///
321    /// // By default all tags are allowed
322    /// assert!(Structex::<regex::Regex>::new(expr).is_ok());
323    ///
324    /// // If only 'B' is allowed then the above expression is invalid
325    /// assert!(
326    ///     StructexBuilder::new(expr)
327    ///         .with_allowed_argless_tags('B')
328    ///         .build::<regex::Regex>()
329    ///         .is_err()
330    /// );
331    /// ```
332    pub fn with_allowed_argless_tags(mut self, tags: impl Into<String>) -> Self {
333        self.allowed_argless_tags = Some(tags.into());
334        self
335    }
336
337    /// This sets the allowed tags when a slash delimited argument is provided.
338    ///
339    /// By default, all tags are allowed but this may be used to cause a compile error if the input
340    /// expression contains tags that were not expected.
341    ///
342    /// # Example
343    /// ```
344    /// use structex::{Structex, StructexBuilder};
345    ///
346    /// let expr = "x/.*foo/ A/foo/";
347    ///
348    /// // By default all tags are allowed
349    /// assert!(Structex::<regex::Regex>::new(expr).is_ok());
350    ///
351    /// // If only 'B' is allowed then the above expression is invalid
352    /// assert!(
353    ///     StructexBuilder::new(expr)
354    ///         .with_allowed_single_arg_tags('B')
355    ///         .build::<regex::Regex>()
356    ///         .is_err()
357    /// );
358    /// ```
359    pub fn with_allowed_single_arg_tags(mut self, tags: impl Into<String>) -> Self {
360        self.allowed_single_arg_tags = Some(tags.into());
361        self
362    }
363
364    /// Compiles the expression passed to [StructexBuilder::new] with the configuration set on this
365    /// builder.
366    ///
367    /// If the expression was invalid, an error is returned.
368    pub fn build<R>(self) -> Result<Structex<R>, Error>
369    where
370        R: Re,
371    {
372        let mut c = Compiler {
373            require_actions: self.require_actions,
374            allowed_argless_tags: self.allowed_argless_tags,
375            allowed_single_arg_tags: self.allowed_single_arg_tags,
376            ..Default::default()
377        };
378
379        let inst = c.compile(&self.expr)?;
380        let Compiler {
381            re, tags, actions, ..
382        } = c;
383
384        let actions: Vec<_> = actions
385            .into_iter()
386            .map(|mut a| {
387                a.arg = a.arg.take().map(|s| (self.action_arg_fn)(s));
388                Arc::new(a)
389            })
390            .collect();
391
392        Ok(Structex {
393            raw: Arc::from(self.expr),
394            inner: Arc::new(Inner {
395                inst,
396                re: re
397                    .into_iter()
398                    .map(|re| R::compile(&re).map_err(|e| Error::Regex(Box::new(e))))
399                    .collect::<Result<Vec<_>, _>>()?,
400                tags,
401                actions,
402            }),
403        })
404    }
405}

417#[derive(Debug, Clone, PartialEq, Eq)]
418pub(super) enum Dot {
419    Range { from: usize, to: usize },
420    Captures(RawCaptures),
421}
422
423impl Dot {
424    pub fn loc(&self) -> (usize, usize) {
425        match self {
426            Self::Range { from, to } => (*from, *to),
427            Self::Captures(caps) => caps.get_match(),
428        }
429    }
430
431    pub fn from(&self) -> usize {
432        match self {
433            Self::Range { from, .. } => *from,
434            Self::Captures(caps) => caps.from(),
435        }
436    }
437
438    pub fn to(&self) -> usize {
439        match self {
440            Self::Range { to, .. } => *to,
441            Self::Captures(caps) => caps.to(),
442        }
443    }
444
445    fn into_captures<'h, R>(self, haystack: &'h R::Haystack) -> Captures<'h, R>
446    where
447        R: Re,
448    {
449        match self {
450            Self::Range { from, to } => Captures::new(haystack, vec![Some((from, to))]),
451            Self::Captures(c) => Captures::new(haystack, c.caps),
452        }
453    }
454}

456/// Represents a tagged capture group for a single match position located by a [Structex].
457///
458/// If an action was specified at the match point in the original [Structex] then `action` will
459/// contain that action, otherwise it will be `None`.
460#[derive(Debug, PartialEq, Eq)]
461pub struct TaggedCaptures<'h, R>
462where
463    R: Re,
464{
465    /// The match and any captures extracted from it.
466    pub captures: Captures<'h, R>,
467    /// An optional [Action] assigned by the match if one was specified in the [Structex]
468    /// expression.
469    pub action: Option<Arc<Action>>,
470}
471
472impl<'h, R> TaggedCaptures<'h, R>
473where
474    R: Re,
475{
476    /// Returns the [slice][Re::Slice] of the haystack that matched.
477    pub fn as_slice(&self) -> &R::Slice {
478        self.captures.match_text()
479    }
480
481    /// Returns the action tag that was associated with the match if one was specified.
482    pub fn tag(&self) -> Option<char> {
483        self.action.as_ref().map(|a| a.tag)
484    }
485
486    /// Returns the argument following the action tag that was associated with the match if one
487    /// was specified.
488    pub fn arg(&self) -> Option<&str> {
489        self.action.as_ref().and_then(|a| a.arg.as_deref())
490    }
491
492    /// Whether or not this match has an assigned [Action].
493    pub fn has_action(&self) -> bool {
494        self.action.is_some()
495    }
496}

518/// An iterator over all matches in a haystack.
519///
520/// This iterator yields [TaggedCaptures] values. The iterator stops when no more matches can be
521/// found. `'h` is the lifetime of the haystack.
522///
523/// This iterator is created by [Structex::iter_tagged_captures].
524pub struct TaggedCapturesIter<'h, R>
525where
526    R: Re,
527{
528    inner: Option<MatchesInner<'h, R>>,
529}

531impl<'h, R> TaggedCapturesIter<'h, R>
532where
533    R: Re,
534{
535    fn new(inst: &'h Inst, inner: Arc<Inner<R>>, haystack: &'h R::Haystack) -> Self {
536        Self {
537            inner: MatchesInner::new(
538                inst,
539                inner,
540                haystack,
541                Dot::Range {

561enum MatchesInner<'h, R>
562where
563    R: Re,
564{
565    Extract(extract::Iter<'h, R>),
566    Parallel(parallel::Iter<'h, R>),
567    Emit(Option<TaggedCaptures<'h, R>>),
568}
569
570impl<'h, R> MatchesInner<'h, R>
571where
572    R: Re,
573{
574    fn new(
575        inst: &'h Inst,
576        inner: Arc<Inner<R>>,
577        haystack: &'h R::Haystack,
578        dot: Dot,
579    ) -> Option<Self> {
580        match inst {
581            // EmitMatch and Action just emit their value
582            Inst::EmitMatch => Some(Self::Emit(Some(TaggedCaptures {
583                captures: dot.into_captures(haystack),
584                action: None,
585            }))),
586            Inst::Action(i) => Some(Self::Emit(Some(TaggedCaptures {
587                captures: dot.into_captures(haystack),
588                action: Some(inner.actions[*i].clone()),
589            }))),
590
591            // Narrow and Guard act as filters on the instructions they wrap
592            Inst::Narrow(n) => n.apply(haystack, dot, inner),
593            Inst::Guard(g) => g.apply(haystack, dot, inner),
594
595            // Extract and Parallel are actual iterators
596            Inst::Extract(ext) => {
597                Some(Self::Extract(extract::Iter::new(haystack, dot, ext, inner)))
598            }
599            Inst::Parallel(bs) => Some(Self::Parallel(parallel::Iter::new(
600                haystack, dot, bs, inner,
601            ))),
602        }
603    }
604}
605
606impl<'h, R> Iterator for MatchesInner<'h, R>
607where
608    R: Re,
609{
610    type Item = TaggedCaptures<'h, R>;
611
612    fn next(&mut self) -> Option<Self::Item> {
613        match self {
614            Self::Extract(ext) => ext.next(),
615            Self::Parallel(p) => p.next(),
616            Self::Emit(opt) => opt.take(),
617        }
618    }
