8#[derive(Debug, Clone, PartialEq, Eq)]
9pub(crate) struct Extract {
10    pub re: usize,
11    pub on_extract: Option<Box<Inst>>,
12    pub on_filter: Option<Box<Inst>>,
13}
14
15impl Extract {
16    /// Try to merge this Extract into an existing branch if possible.
17    ///
18    /// Returns None if merging was successful or Some(self) if not.
19    pub(crate) fn try_merge(self, branches: &mut [Inst]) -> Option<Self> {
20        for branch in branches.iter_mut() {
21            match branch {
22                Inst::Extract(e) if self.re == e.re => match (
23                    e.on_extract.is_none(),
24                    e.on_filter.is_none(),
25                    self.on_extract.is_none(),
26                    self.on_filter.is_none(),
27                ) {
28                    (true, false, false, true) => {
29                        e.on_extract = self.on_extract;
30                        return None;
31                    }
32                    (false, true, true, false) => {
33                        e.on_filter = self.on_filter;
34                        return None;
35                    }
36                    _ => (),
37                },
38                _ => (),
39            }
40        }
41
42        Some(self)
43    }
44}

46pub(super) struct Iter<'h, R>
47where
48    R: Re,
49{
50    haystack: &'h R::Haystack,
51    ext: &'h Extract,
52    inner: Arc<Inner<R>>,
53    /// The original parent dot we are extracting from
54    parent: Dot,
55    /// The child branch we are currently iterating over
56    child: Option<Box<MatchesInner<'h, R>>>,
57    /// The current match
58    held: Option<RawCaptures>,
59    /// The current byte offset we are up to
60    pos: usize,
61}
62
63impl<'h, R> Iter<'h, R>
64where
65    R: Re,
66{
67    pub fn new(
68        haystack: &'h R::Haystack,
69        parent: Dot,
70        ext: &'h Extract,
71        inner: Arc<Inner<R>>,
72    ) -> Self {
73        let pos = parent.from();
74
75        Self {
76            haystack,
77            ext,
78            inner,
79            parent,
80            child: None,
81            held: None,
82            pos,
83        }
84    }
85
86    fn next_captures(&self) -> Option<RawCaptures> {
87        self.inner.re[self.ext.re].captures_between(self.haystack, self.pos, self.parent.to())
88    }
89
90    fn set_extract(&mut self, dot: Dot) {
91        self.child = self.ext.on_extract.as_ref().and_then(|extract| {
92            MatchesInner::new(extract, self.inner.clone(), self.haystack, dot).map(Box::new)
93        });
94    }
95
96    fn set_filter(&mut self, dot: Dot) {
97        self.child = self.ext.on_filter.as_ref().and_then(|filter| {
98            MatchesInner::new(filter, self.inner.clone(), self.haystack, dot).map(Box::new)
99        });
100    }
101}

103impl<'h, R> Iterator for Iter<'h, R>
104where
105    R: Re,
106{
107    type Item = TaggedCaptures<'h, R>;
108
109    fn next(&mut self) -> Option<Self::Item> {
110        loop {
111            // If our current child iterator has something to yield then return that
112            let next = self.child.as_mut().and_then(|child| child.next());
113            if next.is_some() {
114                return next;
115            }
116
117            // If we've reached the end of the input we were given to search then we're done
118            if self.pos == self.parent.to() {
119                return None;
120            }
121
122            // If we have a held match then update child
123            if let Some(caps) = self.held.take() {
124                self.pos = caps.to();
125                self.set_extract(Dot::Captures(caps));
126            } else {
127                // Otherwise find the next match
128                match self.next_captures() {
129                    Some(caps) => {
130                        if self.ext.on_filter.is_some() && caps.from() > self.pos {
131                            self.set_filter(Dot::Range {
132                                from: self.pos,
133                                to: caps.from(),
134                            });
135                            self.pos = caps.from();
136                            self.held = Some(caps);
137                        } else {
138                            self.pos = caps.to();
139                            self.set_extract(Dot::Captures(caps));
140                        }
141                    }
142
143                    None => {
144                        self.set_filter(Dot::Range {
145                            from: self.pos,
146                            to: self.parent.to(),
147                        });
148                        self.pos = self.parent.to();
149                    }
150                }
151            }
152        }
153    }
154}
